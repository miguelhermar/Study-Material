-- create accounts
use role accountadmin; -- superuser in snowflake, can manage all aspects of the account.

create warehouse dbt_wh with warehouse_size = 'x-small';

create database if not exists dbt_db;
create role if not exists dbt_role;

show grants on warehouse dbt_wh;

SHOW ROLES;

grant role dbt_role to user miguelhermar;
grant usage on warehouse dbt_wh to role dbt_role;
grant all on database dbt_db to role dbt_role;

use role dbt_role;

-- schemas are used to logically group objects such as tables, views, and functions within a database
create schema if not exists dbt_db.dbt_schema; 


-- clean up
use role accountadmin;

drop warehouse if exists dbt_wh;
drop database if exists dbt_db;
drop role if exists dbt_role;
